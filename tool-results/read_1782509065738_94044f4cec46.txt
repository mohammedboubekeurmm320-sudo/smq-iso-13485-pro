   126тЖТ   126тЖТfunction getPeriodCutoff(period: string): Date | null {
   127тЖТ   127тЖТ  const now = new Date();
   128тЖТ   128тЖТ  switch (period) {
   129тЖТ   129тЖТ    case '3m': return new Date(now.getFullYear(), now.getMonth() - 3, now.getDate());
   130тЖТ   130тЖТ    case '6m': return new Date(now.getFullYear(), now.getMonth() - 6, now.getDate());
   131тЖТ   131тЖТ    case '12m': return new Date(now.getFullYear(), now.getMonth() - 12, now.getDate());
   132тЖТ   132тЖТ    case 'all': return null;
   133тЖТ   133тЖТ    default: return new Date(now.getFullYear(), now.getMonth() - 12, now.getDate());
   134тЖТ   134тЖТ  }
   135тЖТ   135тЖТ}
   136тЖТ   136тЖТ
   137тЖТ   137тЖТfunction filterByPeriod<T extends { createdAt: string }>(items: T[], period: string): T[] {
   138тЖТ   138тЖТ  const cutoff = getPeriodCutoff(period);
   139тЖТ   139тЖТ  if (!cutoff) return items;
   140тЖТ   140тЖТ  return items.filter(item => new Date(item.createdAt) >= cutoff);
   141тЖТ   141тЖТ}

   143тЖТ   143тЖТfunction getMonthKey(dateStr: string): string {

   146тЖТ   146тЖТ}
