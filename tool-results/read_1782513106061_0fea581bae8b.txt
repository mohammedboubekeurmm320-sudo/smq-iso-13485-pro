   126→function getPeriodCutoff(period: string): Date | null {
   127→  const now = new Date();
   128→  switch (period) {
   129→    case '3m': return new Date(now.getFullYear(), now.getMonth() - 3, now.getDate());
   130→    case '6m': return new Date(now.getFullYear(), now.getMonth() - 6, now.getDate());
   131→    case '12m': return new Date(now.getFullYear(), now.getMonth() - 12, now.getDate());
   132→    case 'all': return null;
   133→    default: return new Date(now.getFullYear(), now.getMonth() - 12, now.getDate());
   134→  }
   135→}
   136→
   137→function filterByPeriod<T extends { createdAt: string }>(items: T[], period: string): T[] {
   138→  const cutoff = getPeriodCutoff(period);
   139→  if (!cutoff) return items;
   140→  return items.filter(item => new Date(item.createdAt) >= cutoff);
   141→}
   142→
   143→function getMonthKey(dateStr: string): string {
   144→  const d = new Date(dateStr);
   145→  return `${d.getFullYear()}-${String(d.getMonth() + 1).padStart(2, '0')}`;
   146→}
   147→
   148→function formatMonthLabel(key: string): string {
   149→  const [year, month] = key.split('-');
   150→  const monthNames = ['Jan', 'Feb', 'Mar', 'Apr', 'May', 'Jun', 'Jul', 'Aug', 'Sep', 'Oct', 'Nov', 'Dec'];
   151→  return `${monthNames[parseInt(month) - 1]} ${year.slice(2)}`;
   152→}
   153→
   154→function pct(numerator: number, denominator: number): number {
   155→  if (denominator === 0) return 0;
   156→  return Math.round((numerator / denominator) * 100);
   157→}

   163→interface CustomTooltipProps {
   164→  active?: boolean;
   165→  payload?: Array<{ name: string; value: number; color?: string }>;
   166→  label?: string;

   169→function CustomTooltip({ active, payload, label }: CustomTooltipProps) {
   170→  if (!active || !payload?.length) return null;
   171→  return (
   172→    <div className="rounded-lg border bg-background p-2 shadow-sm">
   173→      <p className="text-xs font-medium mb-1">{label}</p>
   174→      {payload.map((entry, i) => (
   175→        <p key={i} className="text-xs" style={{ color: entry.color }}>
   176→          {entry.name}: {entry.value}
   177→        </p>
   178→      ))}
   179→    </div>
   180→  );
   181→}

   235→        </div>
   236→      </CardContent>
   237→    </Card>
   238→  );
